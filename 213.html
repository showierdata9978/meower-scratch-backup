Oh boy. It's been a little while. I gave up on manually converting our phrase list into Meower PhraseDB, and wrote a python script to do that for me.

Required files: INPUT.txt
Will return: WORDS.txt, PHRASE_DB.txt

import string

## Step 1: Read contents, split them, and remove punctuation
words = ["@(ausername)"]
sentences = []
with open('INPUT.txt') as f:
    lines = f.read().splitlines()
for x in lines:
    inp = x.split(" ")
    out = []
    for y in inp:
        out.append(str(y))
        if not str(y) in words:
            if not len(str(y)) == 0:
                words.append(str(y))
    out2 = ""
    for z in out:
        out2 = out2 + z + " "
    sentences.append(out2)

## Step 2: Write a new file called "WORDS" containing the contents of words list
wordfile = open("WORDS.txt", "w")
for element in words:
    wordfile.write(str(element))
    wordfile.write("\n")
wordfile.close()

## Step 3: Read back WORDS.txt file and generate a list of phrases in Meower PhraseDB format
with open("WORDS.txt") as f:
    lines = f.read().splitlines()
foobar = []
for f in lines:
    foobar.append(f)
phrases = []
for x in sentences:
    y = x.split()
    z = ""
    for a in y:
        z = z + str(foobar.index(a)+1) + ";"
    if not z in phrases:
        phrases.append(z)
  
## Step 4: Write a new file called "PHRASE_DB" containing the contents of phrases list
wordfile = open("PHRASE_DB.txt", "w")
for element in phrases:
    wordfile.write(str(element))
    wordfile.write("\n")
wordfile.close()
