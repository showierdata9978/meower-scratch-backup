updated close button code with fixes for the br not deleting on the blog and it not working for the first post:
edit: updated updated version: now will work correctly on both main page and blog
function removePost(button) {
	let x = button.parentElement.parentElement.parentElement; 
	let removeBr = function(elem) {if (elem.tagName === "BR") {elem.remove();}}
	if (x.previousElementSibling && x.previousElementSibling.tagName === "BR") {
		let s = x.previousElementSibling;
		removeBr(s);
	} else if (x.nextElementSibling && x.nextElementSibling.tagName === "BR") {
		let s = x.nextElementSibling;
		removeBr(s);
	}
	x.remove();
}
